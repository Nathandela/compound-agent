1//! Vector compatibility spike: validate that the Rust ort crate produces
2//! identical embeddings to the TypeScript Transformers.js implementation
3//! for nomic-embed-text-v1.5 (ONNX Q8 quantized).
4//!
5//! Validates assumptions A2 (vector compatibility) and A6 (ort loads Q8 model).

7use ndarray::{Array1, Array2, Axis};
8use ort::session::Session;
9use serde::Deserialize;
10use std::path::Path;
11use tokenizers::Tokenizer;
12
13/// A reference text/vector pair from the TS implementation.
14#[derive(Deserialize)]
15struct ReferenceEntry {
16    text: String,
17    vector: Vec<f32>,
18}
19
20/// Cosine similarity between two vectors.
21fn cosine_similarity(a: &[f32], b: &[f32]) -> f32 {
22    assert_eq!(a.len(), b.len(), "Vector dimensions must match");
23    let dot: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
24    let mag_a: f32 = a.iter().map(|x| x * x).sum::<f32>().sqrt();
25    let mag_b: f32 = b.iter().map(|x| x * x).sum::<f32>().sqrt();
26    if mag_a == 0.0 || mag_b == 0.0 {
27        return 0.0;
28    }
29    dot / (mag_a * mag_b)
30}
31
32/// Mean pooling: average token embeddings weighted by attention_mask.
33fn mean_pool(token_embeddings: &Array2<f32>, attention_mask: &[i64]) -> Array1<f32> {
34    let (seq_len, hidden_dim) = (token_embeddings.shape()[0], token_embeddings.shape()[1]);
35    let mut pooled = Array1::<f32>::zeros(hidden_dim);
36    let mut mask_sum: f32 = 0.0;
37
38    for i in 0..seq_len {
39        let mask_val = attention_mask[i] as f32;
40        if mask_val > 0.0 {
41            for j in 0..hidden_dim {
42                pooled[j] += token_embeddings[[i, j]] * mask_val;
43            }
44            mask_sum += mask_val;
45        }
46    }
47
48    if mask_sum > 0.0 {
49        pooled /= mask_sum;
50    }
51
52    pooled
53}
54
55/// L2-normalize a vector in place.
56fn l2_normalize(v: &mut Array1<f32>) {
57    let magnitude: f32 = v.iter().map(|x| x * x).sum::<f32>().sqrt();
58    if magnitude > 0.0 {
59        *v /= magnitude;
60    }
61}

63fn main() -> Result<(), Box<dyn std::error::Error>> {
64    // Resolve paths relative to this project's location
65    let project_dir = Path::new(env!("CARGO_MANIFEST_DIR"));
66    let repo_root = project_dir
67        .parent()
68        .unwrap()
69        .parent()
70        .unwrap()
71        .parent()
72        .unwrap();
73
74    let model_path = repo_root.join(
75        "node_modules/.pnpm/@huggingface+transformers@3.8.1/\
76         node_modules/@huggingface/transformers/.cache/\
77         nomic-ai/nomic-embed-text-v1.5/onnx/model_quantized.onnx",
78    );
79    let tokenizer_path = repo_root.join(
80        "node_modules/.pnpm/@huggingface+transformers@3.8.1/\
81         node_modules/@huggingface/transformers/.cache/\
82         nomic-ai/nomic-embed-text-v1.5/tokenizer.json",
83    );
84    let reference_path = project_dir.parent().unwrap().join("reference-vectors.json");
85
86    println!("=== Vector Compatibility Spike ===");
87    println!("Model:     {}", model_path.display());
88    println!("Tokenizer: {}", tokenizer_path.display());
89    println!("Reference: {}", reference_path.display());
90
91    // Verify files exist
92    assert!(model_path.exists(), "ONNX model not found: {}", model_path.display());
93    assert!(tokenizer_path.exists(), "Tokenizer not found: {}", tokenizer_path.display());
94    assert!(reference_path.exists(), "Reference vectors not found: {}", reference_path.display());
95
96    // Load reference data
97    let reference_json = std::fs::read_to_string(&reference_path)?;
98    let references: Vec<ReferenceEntry> = serde_json::from_str(&reference_json)?;
99    println!("\nLoaded {} reference vectors", references.len());
100
101    // Load tokenizer
102    let tokenizer = Tokenizer::from_file(&tokenizer_path)
103        .map_err(|e| format!("Failed to load tokenizer: {}", e))?;
104    println!("[OK] Tokenizer loaded (BertTokenizer)");
105
106    // Load ONNX model via ort
107    let session = Session::builder()?
108        .with_intra_threads(4)?
109        .commit_from_file(&model_path)?;
110    println!("[OK] ONNX model loaded via ort (validates A6)");
111
112    // Print model input/output info
113    println!("\nModel inputs:");
114    for input in session.inputs.iter() {
115        println!("  - {} : {:?}", input.name, input.input_type);
116    }
117    println!("Model outputs:");
118    for output in session.outputs.iter() {
119        println!("  - {} : {:?}", output.name, output.output_type);
120    }
121
122    // Process each reference entry
123    let mut all_pass = true;
124    let mut min_sim: f32 = 1.0;
125    let mut max_sim: f32 = 0.0;
126    let mut total_sim: f64 = 0.0;
127    let mut fail_count = 0;
128
129    println!("\n--- Results ---");
130    println!("{:<4} {:<8} {:<60} {}", "#", "sim", "text (truncated)", "status");
131    println!("{}", "-".repeat(90));
132
133    for (i, entry) in references.iter().enumerate() {
134        // Tokenize
135        let encoding = tokenizer
136            .encode(entry.text.as_str(), true)
137            .map_err(|e| format!("Tokenize failed for '{}': {}", entry.text, e))?;
138
139        let input_ids: Vec<i64> = encoding.get_ids().iter().map(|&id| id as i64).collect();
140        let attention_mask: Vec<i64> = encoding.get_attention_mask().iter().map(|&m| m as i64).collect();
141        let token_type_ids: Vec<i64> = encoding.get_type_ids().iter().map(|&t| t as i64).collect();
142        let seq_len = input_ids.len();
143
144        // Create input tensors with batch dimension [1, seq_len]
145        let input_ids_array = ndarray::Array2::from_shape_vec(
146            (1, seq_len),
147            input_ids.clone(),
148        )?;
149        let attention_mask_array = ndarray::Array2::from_shape_vec(
150            (1, seq_len),
151            attention_mask.clone(),
152        )?;
153        let token_type_ids_array = ndarray::Array2::from_shape_vec(
154            (1, seq_len),
155            token_type_ids,
156        )?;
157
158        // Run inference
159        // The model expects: input_ids, attention_mask, token_type_ids
160        let outputs = session.run(ort::inputs![
161            "input_ids" => input_ids_array,
162            "attention_mask" => attention_mask_array,
163            "token_type_ids" => token_type_ids_array,
164        ]?)?;
165
166        // Extract output tensor: shape [1, seq_len, 768]
167        let output_tensor = outputs[0]
168            .try_extract_tensor::<f32>()?;
169        let output_view = output_tensor.view();
170
171        // Get [seq_len, 768] slice (remove batch dim)
172        let hidden_dim = output_view.shape()[2];
173        let token_embeddings = Array2::from_shape_vec(
174            (seq_len, hidden_dim),
175            output_view
176                .slice(ndarray::s![0, .., ..])
177                .iter()
178                .copied()
179                .collect(),
180        )?;
181
182        // Mean pooling + L2 normalize
183        let mut pooled = mean_pool(&token_embeddings, &attention_mask);
184        l2_normalize(&mut pooled);
185
186        let rust_vec: Vec<f32> = pooled.to_vec();
187
188        // Compare with reference
189        let sim = cosine_similarity(&rust_vec, &entry.vector);
190        let pass = sim > 0.999;
191
192        let truncated_text: String = if entry.text.len() > 57 {
193            format!("{}...", &entry.text[..57])
194        } else {
195            entry.text.clone()
196        };
197        let status = if pass { "PASS" } else { "FAIL" };
198
199        println!(
200            "{:<4} {:.6} {:<60} {}",
201            i + 1,
202            sim,
203            truncated_text,
204            status
205        );
206
207        if !pass {
208            all_pass = false;
209            fail_count += 1;
210            // Print first few dimensions for debugging
211            println!("       Rust[0..5]: {:?}", &rust_vec[..5.min(rust_vec.len())]);
212            println!("       TS  [0..5]: {:?}", &entry.vector[..5.min(entry.vector.len())]);
213        }
214
215        min_sim = min_sim.min(sim);
216        max_sim = max_sim.max(sim);
217        total_sim += sim as f64;
218    }
219
220    let avg_sim = total_sim / references.len() as f64;
221
222    println!("{}", "-".repeat(90));
223    println!("\n=== Summary ===");
224    println!("Total pairs:  {}", references.len());
225    println!("Passed:       {}", references.len() - fail_count);
226    println!("Failed:       {}", fail_count);
227    println!("Min cosine:   {:.6}", min_sim);
228    println!("Max cosine:   {:.6}", max_sim);
229    println!("Avg cosine:   {:.6}", avg_sim);
230
231    if all_pass {
232        println!("\n[PASS] A2 VALIDATED: All {} pairs have cosine_sim > 0.999", references.len());
233        println!("[PASS] A6 VALIDATED: ort crate successfully loaded ONNX Q8 model");
234    } else {
235        println!("\n[FAIL] {} pairs below 0.999 threshold", fail_count);
236        std::process::exit(1);
237    }
238
239    Ok(())
240}
